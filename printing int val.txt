
// Getting input via STDIN
const readline = require("readline");
const inp = readline.createInterface({
  input: process.stdin
  
});
const userInput=[];

inp.on("line", (data) => {
    userInput.push(data);
});

inp.on("close", () => {
   
   //start here
    
  
console.log(userInput.join(' '));
  
  //end here
});
